СОЗДАНИЕ ПРОЭКТА (добавление папки .git в корневую папку с проєктом)
git init

ПРОВЕРКА СТАТУСА 
git status

ДОБАВЛЕНИЕ ВСЕХ СОЗДАНЫХ ФАЙЛОВ 
git add -A

СОЗДАНИЕ КОНТРОЛЬНОЙ ВЕРСИИ
git commit -a -m"НАШ КОМЕНТАРИЙ(краткое описание внесеных изменений)"

ПРОСМОТР ВСЕХ КОМИТОВ (контрольных версий)
git log

ВЫХОД С ТЕКУЩЕЙ ПАПКИ 
cd..

КОПИРОВАНИЕ ПРОЭКТА С GitHub
git clone ссылка скопированная с DitHub название папки в к-рой создастся кнос(папка создается)

ОТПРАВКА ПРОЭКТА НА УДАЛЕННЫЙ РЕПОЗИТОРИЙ (GitHub)
git push

ОБНОВЛЕНИЕ ТЕКУЩЕЙ ВЕРСИИ ПРОЭКТА НА ЛОКАЛЬНОМ РЕПОЗИТОРИИ С УДАЛЕННОГО
git pull
